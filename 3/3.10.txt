x in %rdi, y in %rsi, z in %rdx
arith3:
  orq %rsi, %rdx  # z | y
  sarq $9, %rdx   # (z | y) >> 9
  notq %rdx       # ~((z | y) >> 9)
  movq %rdx, %rax
  subq %rsi, %rax # ~((z | y) >> 9) - y

short arith3(short x, short y, short z) {
  short p1 = z | y; // +
  short p2 = p1 >> 9; // +
  short p3 = ~p2; // +
  short p4 = p3 - y; // -
  return p4;
}
