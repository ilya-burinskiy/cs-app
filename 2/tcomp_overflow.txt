1. Отрицательное переполнение
  -2^w <= x + y < -2^(w-1)
  (-2^w + 2^(w+1)) mod 2^w = 2^w mod 2^w = 0
  (-2^(w-1) + 2^(w+1)) mod 2^w = (2^(w-1) + 2^w) mod 2^w = 2^(w-1)
  0 <= x + y < 2^(w-1)
  0 <= x + y <= 2^(w-1) - 1
  U2T(0) <= U2T(x + y) <= U2T(2^(w-1) - 1)
  0 <= x + y <= 2^(w-1) - 1
2. Все ок
  -2^(w-1) <= x + y < 2^(w - 1)
3. Положительное переполнение
  2^(w-1) <= x + y < 2^w
  2^(w-1) - 2^w <= x + y < 0
  -2^(w-1) <= x + y < 0
